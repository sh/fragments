module.exports.fragments_encryptAesSha256 = function(fragments_crypto, fragments_encryptionAlgorithm) {
  return function(password, textToEncrypt) {
    var cipher, encrypted;
    cipher = fragments_crypto.createCipher(fragments_encryptionAlgorithm, password);
    encrypted = cipher.update(textToEncrypt, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    return encrypted;
  };
};
